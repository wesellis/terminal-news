```go
        "type": "classified_premium",
        "classified_id": "123",
    },

// Webhook handler (automated)
func HandleStripeWebhook(w http.ResponseWriter, r *http.Request) {
    event := stripe.Event{}
    // ... verify webhook signature ...

    if event.Type == "payment_intent.succeeded" {
        classifiedID := event.Data.Object["metadata"]["classified_id"]

        // Auto-activate premium

            UPDATE classifieds
            SET is_premium = TRUE,
                premium_until = NOW() + INTERVAL '60 days'
            WHERE id = $1

    }
```

## Webhook Event Handlers

### All Automated:

```go
switch event.Type {
case "payment_intent.succeeded":
    // Activate premium feature
    activatePremiumFeature(event)

case "payment_intent.payment_failed":
    // Log failure, notify user
    logPaymentFailure(event)

case "customer.subscription.created":
    // New sponsor subscription
    activateSponsorship(event)

case "customer.subscription.updated":
    // Monthly renewal or plan change
    updateSponsorship(event)

case "customer.subscription.deleted":
    // Cancellation
    deactivateSponsorship(event)

case "invoice.payment_succeeded":
    // Send thank you + monthly report
    sendMonthlyReport(event)

case "invoice.payment_failed":
    // Auto-pause service after 3 failed attempts
    pauseService(event)
}
```

### SQL Query (Runs Daily):

```sql
-- Yesterday's revenue
SELECT
    DATE(created_at) AS date,
    SUM(amount) AS total_revenue,
    COUNT(*) FILTER (WHERE payment_type = 'classified_premium') AS premium_count,
    COUNT(*) FILTER (WHERE payment_type = 'boost') AS boost_count,
    COUNT(*) FILTER (WHERE payment_type = 'sponsorship') AS sponsorship_count
FROM payments
WHERE created_at >= NOW() - INTERVAL '1 day'
  AND status = 'succeeded'
GROUP BY DATE(created_at);
```
